CREATE TABLE Sales (
    SaleID INT PRIMARY KEY,
    ProductName VARCHAR(50),
    Category VARCHAR(30),
    Quantity INT,
    Price DECIMAL(10, 2),
    SaleDate DATE
);

select * from sales;
-- Total quantity sold per product
SELECT ProductName, SUM(Quantity) AS TotalSold
FROM Sales
GROUP BY ProductName;
-- Total revenue per category
SELECT Category, SUM(Quantity * Price) AS TotalRevenue
FROM Sales
GROUP BY Category;
-- Average price per product
SELECT ProductName, AVG(Price) AS AvgPrice
FROM Sales
GROUP BY ProductName;
-- Count of sales per date
SELECT SaleDate, COUNT(*) AS NumberOfSales
FROM Sales
GROUP BY SaleDate;
-- Max and min quantity sold per product
SELECT ProductName, MAX(Quantity) AS MaxQty, MIN(Quantity) AS MinQty
FROM Sales
GROUP BY ProductName;
-- Find products with total quantity sold more than 5:
SELECT ProductName, SUM(Quantity) AS TotalQty
FROM Sales
GROUP BY ProductName
HAVING SUM(Quantity) > 5;
-- order by examples
select ProductName,quantity from sales order by quantity desc;
select SaleDate,ProductName from sales order by saledate;
